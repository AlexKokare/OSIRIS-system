def wave_interference(limit, base_primes):
    # Создаем "тишину" (массив нулей)
    canvas = [0] * limit
    
    for p in base_primes:
        # Каждое простое число пускает свою волну (ставит 1 на своем шаге)
        for i in range(p, limit, p):
            canvas[i] = 1 
            
    # Выводим результат как двоичную строку
    # 0 - это "тишина", потенциальное простое число
    return "".join(map(str, canvas))

# Базовые волны: 2, 3, 5
