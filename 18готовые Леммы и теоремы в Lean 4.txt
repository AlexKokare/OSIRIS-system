/--
`PrimeModel` — абстрактная модель вашей конструкции на уровне Lean.

Мы не фиксируем конкретную реализацию `k`-го простого и примориала,
а задаём их как данные вместе с нужными аксиомами/свойствами. Это даёт
возможность сначала перенести логику доказательств, а уже потом заменить
абстрактные поля на конкретные определения из mathlib.
-/
structure PrimeModel where
  p : ℕ → ℕ
  prime_p : ∀ k, Nat.Prime (p k)
  strictMono_p : StrictMono p
  P : ℕ → ℕ
  P_eq : ∀ k, P k = ∏ i in Finset.range k, p i
  branchCount : ℕ → ℕ
  branchCount_eq : ∀ k, branchCount k = Nat.totient (P k)
  residue : (k : ℕ) → Fin (branchCount k) → ℕ
  residue_coprime : ∀ k (j : Fin (branchCount k)), Nat.Coprime (residue k j) (P k)
  residue_injective_mod :
    ∀ k,
      Set.InjOn
        (fun j : Fin (branchCount k) => residue k j % P k)
        Set.univ

namespace PrimeModel

/-- Параметризация ветви: `x_k(T,j) = P_k T + r_{k,j}`. -/
def x (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (T : ℕ) : ℕ :=
  M.P k * T + M.residue k j

/-- Квадратный слой: `p_m^2 ≤ x_k(T,j) < p_(m+1)^2`. -/
def InSquareLayer (M : PrimeModel) (k m : ℕ) (j : Fin (M.branchCount k)) (T : ℕ) : Prop :=
  M.p m ^ 2 ≤ M.x k j T ∧ M.x k j T < M.p (m + 1) ^ 2

/-- Период маски на frozen-layer. -/
def periodModulus (M : PrimeModel) (k m : ℕ) : ℕ :=
  ∏ i in Finset.Icc (k + 1) m, M.p i

/--
`Survives M k j m T` означает, что `x_k(T,j)` не делится ни на один простой
между `p_k` и `p_m`.
-/
def Survives (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m T : ℕ) : Prop :=
  ∀ i ∈ Finset.Icc (k + 1) m, ¬ M.p i ∣ M.x k j T

/-- Индикатор выживания как `0/1`-маска. -/
def mask (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m T : ℕ) : ℕ :=
  if M.Survives k j m T then 1 else 0

/-- Exact-счётчик выживших на префиксе `0,1,...,T-1`. -/
def survivorCount (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m T : ℕ) : ℕ :=
  ((Finset.range T).filter (fun t => M.Survives k j m t)).card

/-- Число допустимых классов на полном периоде. -/
def allowedCountOnPeriod (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m : ℕ) : ℕ :=
  ((Finset.range (M.periodModulus k m)).filter (fun t => M.Survives k j m t)).card

/-- Exact-плотность на полном периоде, рассматриваемая как рациональное число. -/
def density (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m : ℕ) : ℚ :=
  (M.allowedCountOnPeriod k j m : ℚ) / (M.periodModulus k m : ℚ)

end PrimeModel

end PrimeSequenceLean

/-- Если `q > p_k` — простое, то `q` не делит примориал `P_k`. -/
axiom primorial_not_dvd_of_large_prime
  (M : PrimeModel)
  (k q : ℕ)
  (hq : Nat.Prime q)
  (hkq : M.p k < q) :
  ¬ q ∣ M.P k

/-- Следовательно, `P_k` взаимно просто с любым простым `q > p_k`. -/
axiom primorial_coprime_of_large_prime
  (M : PrimeModel)
  (k q : ℕ)
  (hq : Nat.Prime q)
  (hkq : M.p k < q) :
  Nat.Coprime (M.P k) q

/--
Локальная делимость по простому `q > p_k` задаётся ровно одним классом по модулю `q`.
-/
axiom exists_unique_forbidden_class
  (M : PrimeModel)
  (k q m T : ℕ)
  (j : Fin (M.branchCount k))
  (hq : Nat.Prime q)
  (hkq : M.p k < q) :
  ∃! a : Fin q, ∀ n : ℕ, q ∣ M.x k j n ↔ n ≡ a.val [MOD q]

/--
На квадратном слое простота `x_k(T,j)` эквивалентна тому, что оно переживает
ориентированную маску.
-/
axiom prime_iff_survives_on_layer
  (M : PrimeModel)
  (k m T : ℕ)
  (j : Fin (M.branchCount k))
  (hLayer : M.InSquareLayer k m j T) :
  Nat.Prime (M.x k j T) ↔ M.Survives k j m T

/-- Периодичность маски на frozen-layer. -/
axiom mask_periodic
  (M : PrimeModel)
  (k m T : ℕ)
  (j : Fin (M.branchCount k)) :
  M.mask k j m (T + M.periodModulus k m) = M.mask k j m T

/-- Exact CRT-подсчёт числа допустимых классов на одном полном периоде. -/
axiom allowed_count_eq_prod_sub_one
  (M : PrimeModel)
  (k m : ℕ)
  (j : Fin (M.branchCount k)) :
  M.allowedCountOnPeriod k j m = ∏ i in Finset.Icc (k + 1) m, (M.p i - 1)

/-- Exact-плотность как произведение `(1 - 1 / q)` по relevant primes. -/
axiom density_eq_prod
  (M : PrimeModel)
  (k m : ℕ)
  (j : Fin (M.branchCount k)) :
  M.density k j m = ∏ i in Finset.Icc (k + 1) m, (((M.p i - 1 : ℚ) / (M.p i : ℚ)))

end PrimeSequenceLean

### 2.1. Аналитические утверждения

Сюда относятся:

- Мертенс,
- асимптотика `ρ_(k,m) ~ log p_k / log p_m`,

- оценки ранга через `Li^{-1}`.

Например:

- глобальная оценка `|E_{k,j}(T)|`,

Для Lean нужно отдельно фиксировать:

- точную норму ошибки,
- точный класс функций,
- формальную модель сложности,
- все скрытые константы.

Иначе это не теорема, а исследовательская программа.

На уровне `k` каждый узел дерева задаётся парой `(T,j)` и числом

`x_k(T,j) = T P_k + r_(k,j)`.

### Теорема 7.2.1 — Восстановление адреса по рангу `[PROGRAM]`
Для заданного ранга `n` и уровня `k` существует единственный адрес `(T_n^(k), j_n^(k))`, такой что `p_n` находится именно там:

`T_n^(k) = min { T : k + C_k(T+1) ≥ n }`,

`s_n^(k) = n - k - C_k(T_n^(k))`,

`j_n^(k) = min { j : L_k(T_n^(k), j) ≥ s_n^(k) }`.

И затем

`p_n = T_n^(k) P_k + r_(k, j_n^(k))`.

### Лемма 7.2.2 — Двоичная навигация по весам `[PROGRAM]`
Поиск `T_n^(k)` и `j_n^(k)` можно проводить бинарным спуском по накопленным весам `C_k` и `L_k`.

### Лемма 7.3.1 — Связь ранга с логарифмом `[ANALYTIC]`

`rank(x) ≈ x / log x`,

более точно

`rank(x) ≈ Li(x)`,

и потому

`p_n ≈ Li⁻¹(n)`.

### Теорема 7.3.2 — Общая сложность алгоритма `[COND]`
Если `M_k` вычисляется быстро, то общая сложность обратного рангового спуска полиномиальна по `log p_n`.

Точный индикатор выживания не допускает нетривиальной факторизации через обычные аддитивные подгруппы.

### Лемма 8.4.2 — Барьер для линейной спектральной компактности `[EXACT]`
Полный Фурье-спектр означает, что существенное линейное сжатие через характеры невозможно.

### Теорема 8.5.1 — Оценка сложности `[COND]`

## Блок 1. Exact-часть

### Лемма 1.1.1 — Инвертируемость `P_k` по модулю `q` `[EXACT]`
Если `q > p_k` — простое, то `q ∤ P_k`, следовательно `P_k` обратим в `ℤ/qℤ`:

`q > p_k → ∃ P_k⁻¹ mod q, P_k * P_k⁻¹ ≡ 1 [MOD q]`.

### Лемма 1.1.2 — Локальная делимость = один класс `[EXACT]`
Для фиксированных `k`, `j` и простого `q > p_k` существует единственный класс `a_(q,k,j) mod q`, такой что:

`q ∣ x_k(T,j) ↔ T ≡ a_(q,k,j) [MOD q]`,

где `a_(q,k,j) = -r_(k,j) * P_k⁻¹ mod q`.

### Лемма 1.2.1 — Достаточность проверки делителей до `√x` `[EXACT]`
Для `x > 1`:

`x` простое `↔` ни одно простое `q ≤ √x` не делит `x`.

### Лемма 1.2.2 — Frozen-layer `[EXACT]`
Если

`p_m² ≤ x_k(T,j) < p_(m+1)²`,

то

`x_k(T,j)` простое `↔ M_(k,j)^(m)(T) = 1`,

то есть достаточно проверить отсутствие делителей среди простых из `(p_k, p_m]`.

### Лемма 1.3.1 — Периодичность ориентированной маски `[EXACT]`
На слое `m` маска периодична с периодом

`𝓜_(k,m) = ∏_(p_k < q ≤ p_m) q`:

`M_(k,j)^(m)(T + 𝓜_(k,m)) = M_(k,j)^(m)(T)`.

### Лемма 1.3.2 — Ошибка счётчика ограничена периодом `[EXACT]`
Для любого `T ≥ 0`:

`|A_(k,j)^(m)(T) - ρ_(k,m) * T| ≤ 𝓜_(k,m)`.

Эквивалентно:

`|∑_(t < T) (M_(k,j)^(m)(t) - ρ_(k,m))| ≤ 𝓜_(k,m)`.

### Теорема 1.4.1 — Число выживших на полном периоде `[EXACT]`
На одном полном периоде длины `𝓜_(k,m)` число допустимых значений `T` равно

`W_(k,j)^(m) = ∏_(p_k < q ≤ p_m) (q - 1)`.

### Теорема 1.4.2 — Exact-средняя плотность `[EXACT]`

`ρ_(k,m) = W_(k,j)^(m) / 𝓜_(k,m) = ∏_(p_k < q ≤ p_m) (1 - 1/q)`.

Эта плотность не зависит от ветви `j`.

### Следствие 1.4.3 — Exact-линейность счётчика на периодах `[EXACT]`
Для `N` полных периодов:

`A_(k,j)^(m)(N * 𝓜_(k,m)) = N * W_(k,j)^(m) = N * ρ_(k,m) * 𝓜_(k,m)`.

### Лемма 1.5.1 — Ошибка как сумма по слоям `[EXACT]`

`A_(k,j)^(m)(T) = ρ_(k,m) * T + E_(k,j)^(m)(T)`,

где

`E_(k,j)^(m)(T) = ∑_(t=0)^(T-1) (M_(k,j)^(m)(t) - ρ_(k,m))`,

и

`|E_(k,j)^(m)(T)| ≤ 𝓜_(k,m)`.

### Лемма 1.5.2 — Discrepancy периодической функции `[EXACT]`
Если `f` периодична с периодом `P` и имеет нулевое среднее на периоде, а

`D = max_(0 ≤ U < P) |∑_(t < U) f(t)|`,

то

`|∑_(U ≤ t < U + N P) f(t)| ≤ (N + 1) D`.

## Блок 2. Smooth-часть

### Лемма 2.1.3 — Монотонность и гладкость `Φ_k` `[EXACT]`
Функция

`Φ_k(X) = ∫_2^X ρ̃_k(u) du`

монотонно неубывает и кусочно-дифференцируема, причём

`Φ'_k(X) = ρ̃_k(X)`

на интервалах между квадратными слоями.

### Лемма 2.2.2 — Ошибка smooth-аппроксимации на одном слое `[COND]`
Если `x_k(T,j)` лежит в слое `m`, то

`A_(k,j)^(m)(T) = Ã_(k,j)^(m)(T) + E_(k,j)^(m)(T)`,

и

`|E_(k,j)^(m)(T)| ≤ 𝓜_(k,m)`.

### Теорема 2.3.1 — Мертенс для произведения по простым `[ANALYTIC]`
Для `2 ≤ a < b`:

`∏_(a < q ≤ b) (1 - 1/q) = (∏_(q ≤ b) (1 - 1/q)) / (∏_(q ≤ a) (1 - 1/q))`,

и асимптотически

`∏_(q ≤ x) (1 - 1/q) ~ e^(-γ) / log x`.

### Теорема 2.3.2 — Плотность как отношение логарифмов `[ANALYTIC]`

`ρ_(k,m) = ∏_(p_k < q ≤ p_m) (1 - 1/q) ~ log p_k / log p_m`.

### Лемма 2.3.3 — Явная оценка логарифмических плотностей `[ANALYTIC]`

`log ρ_(k,m) = -∑_(p_k < q ≤ p_m) 1/q + O(∑_(p_k < q ≤ p_m) 1/q²)`

и, в частности,

`|log ρ_(k,m) + ∑_(p_k < q ≤ p_m) 1/q| ≤ C / p_k`.

### Лемма 2.4.1 — Связь smooth-clock с логарифмическим интегралом `[ANALYTIC]`

`Φ_k(X) ~ 2 log p_k * Li(X)`.

### Лемма 2.4.2 — Логарифмический интеграл по трансформированным слоям `[ANALYTIC]`

`Ã_(k,j)(T) ≈ (log p_k / P_k) * (Li(P_k T + r_(k,j)) - Li(r_(k,j)))`.

## Блок 3. Мост между exact и smooth

### Лемма 3.1.1 — Декомпозиция глобальной ошибки `[COND]`

`E_(k,j)(T) = A_(k,j)(T) - Ã_(k,j)(T) = ∑_(m ≤ M(T)) E_(k,j)^(m)(T_m)`.

### Теорема 3.1.2 — Грубая оценка суммарной ошибки `[COND]`
Если discrepancy на слое `m` ограничен `D_(k,j)^(m)`, то

`|E_(k,j)(T)| ≤ ∑_(m ≤ M(T)) D_(k,j)^(m) + O(log p_M) * max_m D_(k,j)^(m)`.

В более грубой форме:

`|E_(k,j)(T)| ≤ # {m ≤ M(T)} * D_max`.

### Лемма 3.2.1 — Монотонность smooth-счётчика `[EXACT]`

`Ã_(k,j)(T_2) - Ã_(k,j)(T_1) = ∫_(T_1)^(T_2) ρ̃_k(x_k(t,j)) dt`

и, в частности,

`Ã_(k,j)(T_2) - Ã_(k,j)(T_1) ≥ (log p_k / log x_k(T_2,j)) * (T_2 - T_1)`.

### Теорема 3.2.2 — Локализация истинного ранга `[COND]`
Если `Ã_(k,j)(T_*) = n`, то истинный параметр `T_n` удовлетворяет

`|T_n - T_*| ≤ (|E_(k,j)(T_*)| + O(1)) / ρ_(k,m(T_*))`.

### Лемма 3.2.3 — Условие на ошибку для точной локализации `[COND]`
Чтобы окно локализации имело практический размер, нужно

`E_max / ρ_(k,m) ≤ log x_k(T,j)`.

В частности, если `E_max = O(log p_k)`, то окно локализации мало.

### Лемма 3.3.1 — Разделение по порогу `[HEURISTIC]`
При выборе порога `p_Q` можно разделить exact- и smooth-вклад:

`ρ_k^(hybrid) = (∏_(p_Q < q ≤ p_m) (1 - 1/q)) * ρ_k^(≤Q)`.

### Теорема 3.3.2 — Гибридный счётчик с локальной коррекцией `[PROGRAM]`

`A_k^(hybrid)(T) = A_k^(≤Q)(T) × Ã_k^(>Q)(T)`

в смысле exact-проверки малых простых и smooth-модели для больших.

## Блок 4. Инверсия ранга и восстановление простых

### Лемма 4.1.2 — Корректность взвешенного спуска `[EXACT]`
Если `S(I)` корректно считает число выживших в интервале `I`, то бинарный спуск за `O(log |I|)` находит ровно `n`-й выживший элемент.

### Лемма 4.2.1 — Стартовое приближение через smooth-clock `[COND]`
Стартовая точка `T_0` определяется из

`Ã_(k,j)(T_0) = n`,

то есть

`T_0 = Ã_(k,j)^(-1)(n)`

с контролируемой ошибкой порядка `E_max / ρ_(k,m)`.

### Теорема 4.2.2 — Exact финальная доводка `[COND]`
После стартового приближения `T_0` точный параметр находится пошаговой коррекцией exact-счётчика, причём число шагов ограничено окном локализации:

`#итераций ≤ 2 E_max / ρ_(k,m)`.

### Теорема 4.2.3 — Глобальная процедура восстановления `n`-го простого `[PROGRAM]`
Алгоритм:

1. выбрать уровень `k`,
2. выбрать ветвь `j` и диапазон,
3. выполнить smooth-локализацию,
4. сделать exact-доводку,
5. вернуть

`p_n = x_k(T_n, j) = P_k T_n + r_(k,j)`.

### Лемма 4.2.4 — Сложность процедуры `[COND]`

`Time(findPrime) = O(log² p_n + E_max)`.

### Лемма 4.3.2 — Локальная плотность для парного решета `[HEURISTIC]`
Для пары обычно запрещаются два класса по модулю `q`, поэтому

`ρ_(k,m)^(pair) ≈ ∏_(p_k < q ≤ p_m) (1 - 2/q + O(1/q²))`.

### Теорема 4.3.3 — Восстановление близнецов через парное решето `[PROGRAM]`
Алгоритм `FindPrime` адаптируется к поиску `n`-й пары близнецов заменой одиночной маски на парную и использованием парной плотности.

## Блок 5. Интеграция и финальное оформление

### Теорема 5.1.1 — Основная теорема `[PROGRAM]`
Для каждого `n ≥ 1` существует алгоритм, который за время

`Time(n) = O(log² p_n + log p_k / min_m ρ_(k,m))`

точно вычисляет `n`-е простое число `p_n`.

### Лемма 5.2.1 — Отношение к решету Эратосфена `[COMPARISON]`
Классическое решето ищет все простые до `N` за `O(N log log N)`, а предлагаемый метод для одного `p_n` претендует на `O(log² p_n + error)`.

### Лемма 5.2.2 — Отношение к методу факторизации `[COMPARISON]`
В предлагаемом методе простота проверяется не факторизацией, а проверкой делимости на контролируемый набор простых `q ≤ p_m`.

/--
`PrimeModel` — абстрактная exact-модель примориально-ветвевой конструкции.

На первом этапе мы не привязываемся к конкретной реализации `k`-го простого
и примориала в mathlib, а фиксируем только те данные и свойства, которые
нужны для переноса exact-ядра.
-/
structure PrimeModel where
  p : ℕ → ℕ
  prime_p : ∀ k, Nat.Prime (p k)
  strictMono_p : StrictMono p
  P : ℕ → ℕ
  P_eq : ∀ k, P k = ∏ i in Finset.range k, p i
  branchCount : ℕ → ℕ
  branchCount_eq : ∀ k, branchCount k = Nat.totient (P k)
  residue : (k : ℕ) → Fin (branchCount k) → ℕ
  residue_coprime : ∀ k (j : Fin (branchCount k)), Nat.Coprime (residue k j) (P k)
  residue_injective_mod :
    ∀ k,
      Set.InjOn
        (fun j : Fin (branchCount k) => residue k j % P k)
        Set.univ

namespace PrimeModel

/-- Параметризация ветви: `x_k(T,j) = P_k T + r_{k,j}`. -/
def x (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (T : ℕ) : ℕ :=
  M.P k * T + M.residue k j

/-- Квадратный слой: `p_m^2 ≤ x_k(T,j) < p_{m+1}^2`. -/
def InSquareLayer (M : PrimeModel) (k m : ℕ) (j : Fin (M.branchCount k)) (T : ℕ) : Prop :=
  M.p m ^ 2 ≤ M.x k j T ∧ M.x k j T < M.p (m + 1) ^ 2

/-- Период ориентированной маски на frozen-layer. -/
def periodModulus (M : PrimeModel) (k m : ℕ) : ℕ :=
  ∏ i in Finset.Icc (k + 1) m, M.p i

/--
`Survives M k j m T` означает, что `x_k(T,j)` не делится ни на один простой
между `p_k` и `p_m` включительно.
-/
def Survives (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m T : ℕ) : Prop :=
  ∀ i ∈ Finset.Icc (k + 1) m, ¬ M.p i ∣ M.x k j T

/-- Индикатор выживания как `0/1`-маска. -/
def mask (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m T : ℕ) : ℕ :=
  if M.Survives k j m T then 1 else 0

/-- Exact-счётчик выживших на префиксе `0,1,...,T-1`. -/
def survivorCount (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m T : ℕ) : ℕ :=
  ((Finset.range T).filter (fun t => M.Survives k j m t)).card

/-- Число допустимых классов на одном полном frozen-периоде. -/
def allowedCountOnPeriod (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m : ℕ) : ℕ :=
  ((Finset.range (M.periodModulus k m)).filter (fun t => M.Survives k j m t)).card

/-- Exact-плотность на полном периоде как рациональное число. -/
def density (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m : ℕ) : ℚ :=
  (M.allowedCountOnPeriod k j m : ℚ) / (M.periodModulus k m : ℚ)

end PrimeModel

/-- Число выживших ветвей в блоке `T` на фиксированном слое `m`. -/
def blockWeight (M : PrimeModel) (k m T : ℕ) : ℕ :=
  ∑ j in Finset.univ, M.mask k j m T

/-- Накопленное число выживших до блока `T`. -/
def prefixWeight (M : PrimeModel) (k m T : ℕ) : ℕ :=
  ∑ t in Finset.range T, M.blockWeight k m t

/-- Локальный счётчик выживших ветвей внутри блока `T` до индекса `j`. -/
def localWeight (M : PrimeModel) (k m T : ℕ) (j : Fin (M.branchCount k)) : ℕ :=
  ∑ a in (Finset.univ.filter fun a : Fin (M.branchCount k) => a.1 ≤ j.1), M.mask k a m T

/-- Ранг выжившего узла `(T,j)` на фиксированном уровне и слое. -/
def nodeRank (M : PrimeModel) (k m T : ℕ) (j : Fin (M.branchCount k)) : ℕ :=
  k + M.prefixWeight k m T + M.localWeight k m T j

/-- Адрес узла на уровне `k`. -/
def Address (M : PrimeModel) (k : ℕ) :=
  ℕ × Fin (M.branchCount k)

/-- Значение, соответствующее адресу `(T,j)`. -/
def valueAtAddress (M : PrimeModel) (k : ℕ) (addr : M.Address k) : ℕ :=
  M.x k addr.2 addr.1

/-- Спецификация: адрес `(T,j)` реализует ранг `n`. -/
def IsAddressOfRank
  (M : PrimeModel)
  (k m n : ℕ)
  (addr : M.Address k) : Prop :=
  let T := addr.1
  let j := addr.2
  M.mask k j m T = 1 ∧ M.nodeRank k m T j = n

/-- На уровне спецификации: ранг `n` реализован хотя бы одним адресом. -/
def RankIsRealized (M : PrimeModel) (k m n : ℕ) : Prop :=
  ∃ addr : M.Address k, M.IsAddressOfRank k m n addr

39|def x (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (T : ℕ) : ℕ :=
40|  M.P k * T + M.residue k j

43|def InSquareLayer (M : PrimeModel) (k m : ℕ) (j : Fin (M.branchCount k)) (T : ℕ) : Prop :=
44|  M.p m ^ 2 ≤ M.x k j T ∧ M.x k j T < M.p (m + 1) ^ 2

70|def density (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m : ℕ) : ℚ :=

/--
`PrimeModel` is an abstract exact model of the primorial-branch construction.

At the first stage we do not tie the development to a concrete implementation
of the `k`-th prime or primorial in mathlib. Instead, we record only the data
and properties needed to move the exact core into Lean.
-/
structure PrimeModel where
  p : ℕ → ℕ
  prime_p : ∀ k, Nat.Prime (p k)
  strictMono_p : StrictMono p
  P : ℕ → ℕ
  P_eq : ∀ k, P k = ∏ i in Finset.range k, p i
  branchCount : ℕ → ℕ
  branchCount_eq : ∀ k, branchCount k = Nat.totient (P k)
  residue : (k : ℕ) → Fin (branchCount k) → ℕ
  residue_coprime : ∀ k (j : Fin (branchCount k)), Nat.Coprime (residue k j) (P k)
  residue_injective_mod :
    ∀ k,
      Set.InjOn
        (fun j : Fin (branchCount k) => residue k j % P k)
        Set.univ

namespace PrimeModel

/-- Branch parametrization: `x_k(T,j) = P_k * T + r_{k,j}`. -/
def x (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (T : ℕ) : ℕ :=
  M.P k * T + M.residue k j

/-- The square layer condition `p_m^2 ≤ x_k(T,j) < p_{m+1}^2`. -/
def InSquareLayer (M : PrimeModel) (k m : ℕ) (j : Fin (M.branchCount k)) (T : ℕ) : Prop :=
  M.p m ^ 2 ≤ M.x k j T ∧ M.x k j T < M.p (m + 1) ^ 2

/-- Period modulus of the frozen mask on layer `m`. -/
def periodModulus (M : PrimeModel) (k m : ℕ) : ℕ :=
  ∏ i in Finset.Icc (k + 1) m, M.p i

/--
`Survives M k j m T` means that `x_k(T,j)` is not divisible by any prime
between `p_k` and `p_m` (inclusive at the upper end, exclusive at the lower).
-/
def Survives (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m T : ℕ) : Prop :=
  ∀ i ∈ Finset.Icc (k + 1) m, ¬ M.p i ∣ M.x k j T

/-- Survival indicator as a `0/1` mask. -/
def mask (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m T : ℕ) : ℕ :=
  if M.Survives k j m T then 1 else 0

/-- Exact count of survivors on the prefix `0,1,...,T-1`. -/
def survivorCount (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m T : ℕ) : ℕ :=
  ((Finset.range T).filter (fun t => M.Survives k j m t)).card

/-- Number of allowed residue classes on one full frozen period. -/
def allowedCountOnPeriod (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m : ℕ) : ℕ :=
  ((Finset.range (M.periodModulus k m)).filter (fun t => M.Survives k j m t)).card

/-- Exact density on a full period, viewed as a rational number. -/
def density (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m : ℕ) : ℚ :=
  (M.allowedCountOnPeriod k j m : ℚ) / (M.periodModulus k m : ℚ)

end PrimeModel

end PrimeSequenceLean

/-- If `q > p_k` is prime, then `q` does not divide the primorial `P_k`. -/
axiom primorial_not_dvd_of_large_prime
  (M : PrimeModel)
  (k q : ℕ)
  (hq : Nat.Prime q)
  (hkq : M.p k < q) :
  ¬ q ∣ M.P k

/-- Therefore `P_k` is coprime to every prime `q > p_k`. -/
axiom primorial_coprime_of_large_prime
  (M : PrimeModel)
  (k q : ℕ)
  (hq : Nat.Prime q)
  (hkq : M.p k < q) :
  Nat.Coprime (M.P k) q

/--
Local divisibility by a prime `q > p_k` is controlled by exactly one residue
class modulo `q`.
-/
axiom exists_unique_forbidden_class
  (M : PrimeModel)
  (k q : ℕ)
  (j : Fin (M.branchCount k))
  (hq : Nat.Prime q)
  (hkq : M.p k < q) :
  ∃! a : Fin q, ∀ n : ℕ, q ∣ M.x k j n ↔ n ≡ a.val [MOD q]

/-- On a square layer, primality is equivalent to survival in the oriented mask. -/
axiom prime_iff_survives_on_layer
  (M : PrimeModel)

  (j : Fin (M.branchCount k))
  (hLayer : M.InSquareLayer k m j T) :
  Nat.Prime (M.x k j T) ↔ M.Survives k j m T

/-- The oriented mask is periodic on a frozen layer. -/
axiom mask_periodic
  (M : PrimeModel)
  (k m T : ℕ)
  (j : Fin (M.branchCount k)) :

/-- Exact CRT count of allowed classes on one full period. -/
axiom allowed_count_eq_prod_sub_one
  (M : PrimeModel)
  (k m : ℕ)
  (j : Fin (M.branchCount k)) :

  (k m : ℕ)
  (j : Fin (M.branchCount k)) :
  M.density k j m = ∏ i in Finset.Icc (k + 1) m, (((M.p i - 1 : ℚ) / (M.p i : ℚ)))

  (M : PrimeModel)
  (k : ℕ)
  (j : Fin (M.branchCount k))
  (m T : ℕ) : ℚ :=
  (M.survivorCount k j m T : ℚ) - M.density k j m * (T : ℚ)

/-- Number of surviving branches in block `T` on a fixed layer `m`. -/
def blockWeight (M : PrimeModel) (k m T : ℕ) : ℕ :=
  ∑ j in Finset.univ, M.mask k j m T

/-- Accumulated number of surviving nodes before block `T`. -/
def prefixWeight (M : PrimeModel) (k m T : ℕ) : ℕ :=
  ∑ t in Finset.range T, M.blockWeight k m t

/-- Local count of surviving branches inside block `T` up to index `j`. -/
def localWeight (M : PrimeModel) (k m T : ℕ) (j : Fin (M.branchCount k)) : ℕ :=
  ∑ a in (Finset.univ.filter fun a : Fin (M.branchCount k) => a.1 ≤ j.1), M.mask k a m T

/-- Rank of a surviving node `(T,j)` on a fixed level and layer. -/
def nodeRank (M : PrimeModel) (k m T : ℕ) (j : Fin (M.branchCount k)) : ℕ :=
  k + M.prefixWeight k m T + M.localWeight k m T j

/-- Address of a node at level `k`. -/
def Address (M : PrimeModel) (k : ℕ) :=
  ℕ × Fin (M.branchCount k)

/-- Value attached to the address `(T,j)`. -/
def valueAtAddress (M : PrimeModel) (k : ℕ) (addr : M.Address k) : ℕ :=
  M.x k addr.2 addr.1

/-- Specification: an address realizes rank `n`. -/
def IsAddressOfRank
  (M : PrimeModel)
  (k m n : ℕ)
  (addr : M.Address k) : Prop :=
  let T := addr.1
  let j := addr.2
  M.mask k j m T = 1 ∧ M.nodeRank k m T j = n

/-- Specification-level existence of an address realizing rank `n`. -/
def RankIsRealized (M : PrimeModel) (k m n : ℕ) : Prop :=
  ∃ addr : M.Address k, M.IsAddressOfRank k m n addr

/--

  instFintype : Fintype Carrier
  instAddCommGroup : AddCommGroup Carrier
  indicator : Carrier → ℤ
  isSurvivalIndicator : Prop

/-- Stabilizer of the indicator under translation. -/
def Stabilizer (C : CompressionModel) : Set C.Carrier :=
  {h | ∀ x, C.indicator (h + x) = C.indicator x}

## File layout

- `PrimeSequenceLean/Foundations.lean` — abstract exact objects, branches, layers, masks, counts, density.

- `PrimeSequenceLean/Barriers.lean` — abstract shell for negative compression/Fourier results.
- `PrimeSequenceLean/Registry.lean` — registry of theorem names and formalization status.

## Why this split matters

The uploaded text mixes several different kinds of mathematics:

1. exact combinatorial sieve statements,
2. analytic asymptotics,
3. conditional error transport,
4. algorithmic recovery claims,
5. negative structural results.

Trying to formalize all of that in one undivided layer would make the project
hard to maintain and mathematically ambiguous. The current split keeps the exact
core clean and leaves later layers explicit about their status.

/--
`PrimeModel` is an abstract exact model of the primorial-branch construction.

На первом этапе мы не привязываем разработку к конкретной реализации `k`-го
простого или primorial в mathlib, а фиксируем только данные и свойства,
необходимые для переноса exact-ядра в Lean.
-/
structure PrimeModel where
  p : ℕ → ℕ
  prime_p : ∀ k, Nat.Prime (p k)
  strictMono_p : StrictMono p
  P : ℕ → ℕ
  P_eq : ∀ k, P k = ∏ i in Finset.range k, p i
  branchCount : ℕ → ℕ
  branchCount_eq : ∀ k, branchCount k = Nat.totient (P k)
  residue : (k : ℕ) → Fin (branchCount k) → ℕ
  residue_coprime : ∀ k (j : Fin (branchCount k)), Nat.Coprime (residue k j) (P k)
  residue_injective_mod :
    ∀ k,
      Set.InjOn
        (fun j : Fin (branchCount k) => residue k j % P k)
        Set.univ

namespace PrimeModel

/-- Branch parametrization: `x_k(T,j) = P_k * T + r_{k,j}`. -/
def x (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (T : ℕ) : ℕ :=
  M.P k * T + M.residue k j

/-- The square layer condition `p_m^2 ≤ x_k(T,j) < p_{m+1}^2`. -/
def InSquareLayer (M : PrimeModel) (k m : ℕ) (j : Fin (M.branchCount k)) (T : ℕ) : Prop :=
  M.p m ^ 2 ≤ M.x k j T ∧ M.x k j T < M.p (m + 1) ^ 2

/-- Period modulus of the oriented mask on layer `m`. -/
def periodModulus (M : PrimeModel) (k m : ℕ) : ℕ :=
  ∏ i in Finset.Icc (k + 1) m, M.p i

/--
`Survives M k j m T` means that `x_k(T,j)` is not divisible by any prime
between `p_k` and `p_m`.
-/
def Survives (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m T : ℕ) : Prop :=
  ∀ i ∈ Finset.Icc (k + 1) m, ¬ M.p i ∣ M.x k j T

/-- Survival indicator as a `0/1` mask. -/
def mask (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m T : ℕ) : ℕ :=
  if M.Survives k j m T then 1 else 0

/-- Exact count of survivors on the prefix `0,1,...,T-1`. -/
def survivorCount (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m T : ℕ) : ℕ :=
  ((Finset.range T).filter (fun t => M.Survives k j m t)).card

/-- Number of allowed residue classes on one full frozen period. -/
def allowedCountOnPeriod (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m : ℕ) : ℕ :=
  ((Finset.range (M.periodModulus k m)).filter (fun t => M.Survives k j m t)).card

/-- Exact density on a full period, viewed as a rational number. -/
def density (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m : ℕ) : ℚ :=
  (M.allowedCountOnPeriod k j m : ℚ) / (M.periodModulus k m : ℚ)

@[simp] theorem x_zero (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) :
    M.x k j 0 = M.residue k j := by
  simp [x]

theorem mask_eq_one_of_survives
    (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m T : ℕ)
    (h : M.Survives k j m T) :
    M.mask k j m T = 1 := by
  simp [mask, h]

theorem mask_eq_zero_of_not_survives
    (M : PrimeModel) (k : ℕ) (j : Fin (M.branchCount k)) (m T : ℕ)
    (h : ¬ M.Survives k j m T) :
    M.mask k j m T = 0 := by
  simp [mask, h]

end PrimeModel

end PrimeSequenceLean

/-- If `q > p_k` is prime, then `q` does not divide the primorial `P_k`. -/
theorem primorial_not_dvd_of_large_prime
  (M : PrimeModel)
  (k q : ℕ)
  (hq : Nat.Prime q)
  (hkq : M.p k < q) :
  ¬ q ∣ M.P k := by
  rw [M.P_eq]
  exact hq.not_dvd_finset_prod fun i hi => by
    intro hdiv
    have hipk : i < k := Finset.mem_range.mp hi
    have hipq : M.p i < q := lt_trans (M.strictMono_p hipk) hkq
    have heq : q = M.p i := (Nat.prime_dvd_prime_iff_eq hq (M.prime_p i)).mp hdiv
    exact (ne_of_lt hipq) heq.symm

/-- Therefore `P_k` is coprime to every prime `q > p_k`. -/
theorem primorial_coprime_of_large_prime
  (M : PrimeModel)
  (k q : ℕ)
  (hq : Nat.Prime q)
  (hkq : M.p k < q) :
  Nat.Coprime (M.P k) q := by
  exact (hq.coprime_iff_not_dvd).2 (primorial_not_dvd_of_large_prime M k q hq hkq) |>.symm

/--
Local divisibility by a prime `q > p_k` is controlled by exactly one residue
class modulo `q`.
-/
axiom exists_unique_forbidden_class
  (M : PrimeModel)
  (k q : ℕ)
  (j : Fin (M.branchCount k))
  (hq : Nat.Prime q)
  (hkq : M.p k < q) :
  ∃! a : Fin q, ∀ n : ℕ, q ∣ M.x k j n ↔ n ≡ a.val [MOD q]

/-- On a square layer, primality is equivalent to survival in the oriented mask. -/
axiom prime_iff_survives_on_layer
  (M : PrimeModel)
  (k m T : ℕ)
  (j : Fin (M.branchCount k))
  (hLayer : M.InSquareLayer k m j T) :
  Nat.Prime (M.x k j T) ↔ M.Survives k j m T

/-- The oriented mask is periodic on a frozen layer. -/
axiom mask_periodic
  (M : PrimeModel)
  (k m T : ℕ)
  (j : Fin (M.branchCount k)) :
  M.mask k j m (T + M.periodModulus k m) = M.mask k j m T

/-- Exact CRT count of allowed classes on one full period. -/
axiom allowed_count_eq_prod_sub_one
  (M : PrimeModel)
  (k m : ℕ)
  (j : Fin (M.branchCount k)) :
  M.allowedCountOnPeriod k j m = ∏ i in Finset.Icc (k + 1) m, (M.p i - 1)

/-- Exact density as the product `∏ (1 - 1/q)` over relevant primes. -/
axiom density_eq_prod
  (M : PrimeModel)
  (k m : ℕ)
  (j : Fin (M.branchCount k)) :
  M.density k j m = ∏ i in Finset.Icc (k + 1) m, (((M.p i - 1 : ℚ) / (M.p i : ℚ)))

/-- Error of the exact counter against the exact mean density on layer `m`. -/
def exactError
  (M : PrimeModel)
  (k : ℕ)
  (j : Fin (M.branchCount k))
  (m T : ℕ) : ℚ :=
  (M.survivorCount k j m T : ℚ) - M.density k j m * (T : ℚ)

/-- On full periods, the exact survivor counter is linear. -/
axiom survivorCount_linear_on_periods
  (M : PrimeModel)
  (k m N : ℕ)
  (j : Fin (M.branchCount k)) :
  M.survivorCount k j m (N * M.periodModulus k m) = N * M.allowedCountOnPeriod k j m

/-- The exact error term is bounded by the period length. -/
axiom exactError_bound
  (M : PrimeModel)
  (k : ℕ)
  (j : Fin (M.branchCount k))
  (m T : ℕ) :
  |M.exactError k j m T| ≤ (M.periodModulus k m : ℚ)

/-- Every prefix inside a period inherits the same discrepancy control. -/
axiom discrepancy_period_bound
  (M : PrimeModel)
  (k : ℕ)
  (j : Fin (M.branchCount k))
  (m U : ℕ)
  (hU : U < M.periodModulus k m) :
  |M.exactError k j m U| ≤ (M.periodModulus k m : ℚ)

@[simp] theorem exactError_def
  (M : PrimeModel)
  (k : ℕ)
  (j : Fin (M.branchCount k))
  (m T : ℕ) :
  M.exactError k j m T = (M.survivorCount k j m T : ℚ) - M.density k j m * (T : ℚ) := rfl

end PrimeSequenceLean

/-- Number of surviving branches in block `T` on a fixed layer `m`. -/
def blockWeight (M : PrimeModel) (k m T : ℕ) : ℕ :=
  ∑ j in Finset.univ, M.mask k j m T

/-- Accumulated number of surviving nodes before block `T`. -/
def prefixWeight (M : PrimeModel) (k m T : ℕ) : ℕ :=
  ∑ t in Finset.range T, M.blockWeight k m t

/-- Local count of surviving branches inside block `T` up to index `j`. -/
def localWeight (M : PrimeModel) (k m T : ℕ) (j : Fin (M.branchCount k)) : ℕ :=
  ∑ a in (Finset.univ.filter fun a : Fin (M.branchCount k) => a.1 ≤ j.1), M.mask k a m T

/-- Rank of a surviving node `(T,j)` on a fixed level and layer. -/
def nodeRank (M : PrimeModel) (k m T : ℕ) (j : Fin (M.branchCount k)) : ℕ :=
  k + M.prefixWeight k m T + M.localWeight k m T j

/-- Address of a node at level `k`. -/
def Address (M : PrimeModel) (k : ℕ) :=
  ℕ × Fin (M.branchCount k)

/-- Value attached to the address `(T,j)`. -/
def valueAtAddress (M : PrimeModel) (k : ℕ) (addr : M.Address k) : ℕ :=
  M.x k addr.2 addr.1

/-- Specification: an address realizes rank `n`. -/
def IsAddressOfRank
  (M : PrimeModel)
  (k m n : ℕ)
  (addr : M.Address k) : Prop :=
  let T := addr.1
  let j := addr.2
  M.mask k j m T = 1 ∧ M.nodeRank k m T j = n

/-- Specification-level existence of an address realizing rank `n`. -/
def RankIsRealized (M : PrimeModel) (k m n : ℕ) : Prop :=
  ∃ addr : M.Address k, M.IsAddressOfRank k m n addr

@[simp] theorem valueAtAddress_mk
  (M : PrimeModel)
  (k T : ℕ)
  (j : Fin (M.branchCount k)) :
  M.valueAtAddress k (T, j) = M.x k j T := rfl

theorem rankIsRealized_iff
  (M : PrimeModel)
  (k m n : ℕ) :
  M.RankIsRealized k m n ↔ ∃ addr : M.Address k, M.IsAddressOfRank k m n addr := Iff.rfl

end PrimeSequenceLean

# PrimeSequenceLean — статус формализации

Этот каталог собирает материал в нормальную структуру Lean 4-проекта.

## Что уже перенесено

- `PrimeSequenceLean/Foundations.lean` — базовые объекты, ветви, слои, маска, счётчики.
- `PrimeSequenceLean/ExactCore.lean` — forbidden class, frozen-layer, периодичность, CRT-плотность, exact-error.
- `PrimeSequenceLean/Rank.lean` — блочные веса, локальные веса, адреса и ранги.
- `PrimeSequenceLean/Smooth.lean` — smooth-слой как отдельный интерфейс.
- `PrimeSequenceLean/Bridge.lean` — мост между exact и smooth.
- `PrimeSequenceLean/Recovery.lean` — восстановление адреса и значения по рангу.
- `PrimeSequenceLean/Barriers.lean` — каркас для барьерных результатов о сжатии.
- `PrimeSequenceLean/Registry.lean` — реестр имён и статусов утверждений.

## Что сделано по сути

1. Разделён exact-слой и аналитический слой.
2. Вынесены базовые определения и спецификации в отдельные модули.
3. Первые две леммы exact-части переведены из `axiom` в реальные Lean-теоремы:
   - `primorial_not_dvd_of_large_prime`
   - `primorial_coprime_of_large_prime`
4. Остальные нетривиальные пункты пока оставлены как `axiom`, чтобы не смешивать каркас с ещё не перенесённой аналитикой.
