/-- Фазовая координата: σ_p(n) = {n/p} ∈ ℚ ∩ [0,1) -/
def sigma (n p : ℕ) : ℚ :=
  if p = 0 then 0 else (n % p : ℚ) / p

/-- Множество простых ≤ √n (горизонт проверки) -/
def small_primes (n : ℕ) : Finset ℕ :=
  (Finset.range (Nat.sqrt n + 1)).filter Nat.Prime

/-- Путевая энергия: сумма весов теневых осей -/
def path_energy (n : ℕ) : ℚ :=
  ∑ p in small_primes n, if n % p = 0 then 1 / (p : ℚ) else 0

/-- Теневая матрица: диагональный проектор на делители ≤ √n -/
def shadow_matrix (n : ℕ) : Matrix (Fin (small_primes n).card) (Fin (small_primes n).card) ℚ :=
  fun i j =>
    if i = j then
      let p := (small_primes n).toList.get ⟨i, by simp; exact i.isLt⟩
      if n % p = 0 then 1 else 0
    else 0

/-- Лемма: Простота ⇔ нулевая путевая энергия -/
theorem prime_iff_path_energy_zero (n : ℕ) (hn : 2 ≤ n) :
  Nat.Prime n ↔ path_energy n = 0 := by
  unfold path_energy
  constructor
  · -- (→) Простое ⇒ нет теней ⇒ энергия = 0
    intro h_prime
    apply Finset.sum_eq_zero
    intro p hp
    have h_p_prime : p.Prime := (Finset.mem_filter.mp hp).2
    have h_p_le : p ≤ Nat.sqrt n := by have := (Finset.mem_filter.mp hp).1; simp at this; exact Nat.le_of_lt_succ this
    have h_not_dvd : ¬ p ∣ n := by
      intro h_dvd
      exact Nat.not_prime_of_dvd_of_lt h_dvd (by omega) h_prime (by omega)
    simp [h_not_dvd, Nat.mod_eq_zero_of_dvd]
  · -- (←) Энергия = 0 ⇒ нет теней ⇒ простое
    intro h_zero
    rw [Nat.prime_def_lt]
    constructor
    · exact hn
    · intro m hm1 hm2
      by_contra h_div
      have h_m_le_sqrt : m ≤ Nat.sqrt n := by
        by_contra h; push_neg at h; have : m*m > n := by nlinarith
        have h_le : m ≤ n/m := Nat.le_of_dvd (by omega) h_div; nlinarith
      have h_m_in : m ∈ small_primes n := by
        simp [small_primes, hm1, h_m_le_sqrt]
        exact Nat.prime_of_dvd_of_dvd m (Nat.prime_of_dvd_of_dvd m h_div)
      have h_pos : path_energy n > 0 := by
        apply Finset.sum_pos
        · intro q hq
          by_cases h : n % q = 0 <;> simp [h]; positivity
        · use m; simp [h_m_in, Nat.mod_eq_zero_of_dvd h_div]; positivity
      linarith

/-- Диагональная матрица весов p^{-s} (дискретный аналог частот ζ) -/
def zeta_weights (s : ℕ) (n : ℕ) : Matrix (Fin (small_primes n).card) (Fin (small_primes n).card) ℚ :=
  fun i j =>
    if i = j then
      let p := (small_primes n).toList.get ⟨i, by simp; exact i.isLt⟩
      if s = 0 then 1 else (1 : ℚ) / (p ^ s : ℚ)
    else 0

/-- Спектральный определитель OSIRIS: det(I - S_n · W_s) -/
noncomputable def osiris_zeta_det (n s : ℕ) : ℚ :=
  Matrix.det (1 - shadow_matrix n * zeta_weights s n)

  have h_det_eq_prod : Matrix.det (1 - shadow_matrix n * zeta_weights s n) =
    ∏ i : Fin (small_primes n).card, (1 - shadow_matrix n * zeta_weights s n) i i := by
    rw [Matrix.det_of_diagonal h_diag, Finset.prod_univ]

  intro i _
  simp [Matrix.one_apply, Matrix.mul_apply, shadow_matrix, zeta_weights]

    · simp [h_div, h_eq]
