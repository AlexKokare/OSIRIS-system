  have h_p_min_prop : start < p_min ∧ Nat.Prime p_min := Nat.find_spec h_exists

  have h_min : ∀ q, start < q → q < p_min → ¬ Nat.Prime q := by

    intro q hq1 hq2

    by_contra h_prime

    have h_lt_find : q < Nat.find h_exists := hq2

    exact Nat.not_lt.mpr h_lt_find (Nat.find_min' h_exists ⟨hq1, h_prime⟩)

  -- search достигает p_min

  have h_eq : (let rec search (n : ℕ) : ℕ := if is_prime_check n then n else search (n + 1); search (start + 1)) = p_min
:=

    search_reaches_min_prime start p_min h_p_min_prop h_min

  rw [h_eq]

  exact h_p_min_prop
