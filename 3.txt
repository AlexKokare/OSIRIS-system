/-- Фазовая координата: σ_p(n) = {n/p} -/
def sigma (n p : ℕ) : ℚ :=
  if p = 0 then 0 else (n % p : ℚ) / (p : ℚ)

/-- Оператор выворота на оси p: V_p(x) = 1-x (инволюция) -/
def eversion (x : ℚ) : ℚ := if x = 0 then 0 else 1 - x

/-- Множество простых ≤ √n -/
def smallPrimes (n : ℕ) : Finset ℕ :=
  (Finset.range (Nat.sqrt n + 1)).filter Nat.Prime

/-- Голографическая энергия H(n) -/
noncomputable def H (n : ℕ) : ℚ :=
  ∑ p in smallPrimes n, (sigma n p - 1/2)^2 * (1 / (p : ℚ))

/-- Путевая энергия (тени делителей) -/
def pathEnergy (n : ℕ) : ℚ :=
  ∑ p in smallPrimes n, if n % p = 0 then 1 / (p : ℚ) else 0

/-- Потолок окна решета -/
noncomputable def Mbound (k : ℕ) : ℚ :=
  ∑ p in (Finset.range (k + 1)).filter Nat.Prime, 1 / (4 * (p : ℚ))

/-- Идеальная голографическая энергия H^*(n) -/
noncomputable def Hstar (n : ℕ) : ℚ :=
  Mbound (Nat.sqrt n) - (1/4) * pathEnergy n

def sigma (n p : ℕ) : ℚ :=
  if p = 0 then 0 else (n % p : ℚ) / (p : ℚ)

def eversion (p : ℕ) (x : ℚ) : ℚ :=
  if x = 0 then 0 else 1 - x

def bit2 (n : ℕ) : ℕ := if n % 2 = 0 then 0 else 1

/-- Пятеричная цифра выворота: остаток mod 5 -/
def digit5 (n : ℕ) : ℕ := n % 5

/-- Код выверта C_{2,5}(n) -/
structure EversionCode25 where
  b : ℕ
  q : ℕ
  b_valid : b = 0 ∨ b = 1
  q_valid : q < 5

def code25 (n : ℕ) : EversionCode25 :=
  { b := bit2 n, q := digit5 n,
    b_valid := by simp [bit2]; split_ifs <;> simp,
    q_valid := Nat.mod_lt n (by decide) }

/-- Действие выворота на код: (b, q) ↦ (b, 5-q mod 5) -/
def eversionOnCode (c : EversionCode25) : EversionCode25 :=
  { b := c.b, q := (5 - c.q) % 5,
    b_valid := c.b_valid,
    q_valid := by simp [Nat.mod_lt]; omega }

/-- Лемма: выворот инволютивен на коде -/
theorem eversion_code_involutive (c : EversionCode25) :
    eversionOnCode (eversionOnCode c) = c := by
  ext <;> simp [eversionOnCode, Nat.mod_eq_of_lt]
  <;> omega

      rcases (hp.eq_one_or_self_of_dvd 5 h) with h1 | h1 <;> omega
    simp [h5, Nat.mod_eq_zero_of_dvd]

def sieve_primes(limit: int) -> list[int]:

    """Решето Эратосфена для получения всех простых чисел до limit."""

    if limit < 2:

        return []

    is_prime = [True] * (limit + 1)

    is_prime[0] = is_prime[1] = False

    for i in range(2, int(limit**0.5) + 1):

        if is_prime[i]:

            for j in range(i * i, limit + 1, i):

                is_prime[j] = False

    return [i for i, p in enumerate(is_prime) if p]



def sigma(n: int, p: int) -> float:

    """Фазовая координата σ_p(n) = (n mod p) / p."""

    if p == 0:

        return 0.0

    return (n % p) / p



def eversion(p: int, x: float) -> float:

    """Оператор выворота V_p(x)."""

    return 0.0 if x == 0.0 else 1.0 - x



def code25(n: int) -> tuple[int, int]:

    """Код выверта C_{2,5}(n) = (b, q)."""

    b = 1 if n % 2 != 0 else 0

    q = n % 5

    return b, q



def path_energy(n: int, primes: list[int]) -> int:

    """

    Путевая энергия E(n): количество теней (нулевых фаз) для p <= sqrt(n).

    E(n) = 0 ⇔ число не имеет делителей среди простых <= sqrt(n).

    """

    limit = int(math.isqrt(n))

    shadows = 0

    for p in primes:

        if p > limit:

            break

        if sigma(n, p) == 0.0:

            shadows += 1

    return shadows



def holographic_energy(n: int, primes: list[int]) -> int:

    """

    Голографическая энергия H*(n) = M(sqrt(n)) - E(n),

    где M(k) — количество простых <= k.

    """

    limit = int(math.isqrt(n))

    M = sum(1 for p in primes if p <= limit)

    return M - path_energy(n, primes)



def prime_probability(n: int, primes: list[int]) -> float:

    """

    Вероятность просвета P_prime(n) = prod_{p <= sqrt(n)} (1 - δ_{σ_p(n), 0}).

    В дискретной модели принимает значения 1.0 (простое) или 0.0 (составное).

    """

    limit = int(math.isqrt(n))

    for p in primes:

        if p > limit:

            break

        if sigma(n, p) == 0.0:

            return 0.0

    return 1.0

import Mathlib.Data.Nat.Prime.Basic

import Mathlib.Data.Rat.Defs

import Mathlib.Data.Nat.Sqrt

import Mathlib.Tactic



namespace OSIRIS.EversionCode



/-- Фазовая координата σ_p(n) -/

def sigma (n p : ℕ) : ℚ :=

  if p = 0 then 0 else (n % p : ℚ) / (p : ℚ)



/-- Оператор выворота на оси p -/

def eversion (p : ℕ) (x : ℚ) : ℚ :=

  if x = 0 then 0 else 1 - x



/-- Двоичный бит выворота: 1 ⇔ нечётность (просвет), 0 ⇔ чётность (тень) -/

def bit2 (n : ℕ) : ℕ := if n % 2 = 0 then 0 else 1



/-- Пятеричная цифра выворота: остаток mod 5 -/

def digit5 (n : ℕ) : ℕ := n % 5



/-- Код выверта C_{2,5}(n) -/

structure EversionCode25 where

  b : ℕ

  q : ℕ

  b_valid : b = 0 ∨ b = 1

  q_valid : q < 5



def code25 (n : ℕ) : EversionCode25 :=

  { b := bit2 n, q := digit5 n,

    b_valid := by simp [bit2]; split_ifs <;> simp,

    q_valid := Nat.mod_lt n (by decide) }



/-- Действие выворота на код: (b, q) ↦ (b, 5-q mod 5) -/

def eversionOnCode (c : EversionCode25) : EversionCode25 :=

  { b := c.b, q := (5 - c.q) % 5,

    b_valid := c.b_valid,

    q_valid := by simp [Nat.mod_lt]; omega }



/-- Лемма: выворот инволютивен на коде -/

theorem eversion_code_involutive (c : EversionCode25) :

    eversionOnCode (eversionOnCode c) = c := by

  ext <;> simp [eversionOnCode, Nat.mod_eq_of_lt]

  <;> omega



/-- Лемма: для простого p > 5 код не содержит теней -/

theorem prime_code_no_shadows (p : ℕ) (hp : p.Prime) (hp5 : 5 < p) :

    (code25 p).b = 1 ∧ (code25 p).q ≠ 0 := by

  unfold code25 bit2 digit5

  constructor

  · -- p нечётно

    have h2 : ¬ 2 ∣ p := fun h => by

      rcases (hp.eq_one_or_self_of_dvd 2 h) with h1 | h1 <;> omega

    simp [h2]

  · -- p не делится на 5

    have h5 : ¬ 5 ∣ p := fun h => by

      rcases (hp.eq_one_or_self_of_dvd 5 h) with h1 | h1 <;> omega

    simp [h5, Nat.mod_eq_zero_of_dvd]



/-- Теорема: код выверта рекурсивно характеризует простоту -/

theorem eversion_code_prime_criterion (n : ℕ) (hn : 5 < n) :

  Nat.Prime n ↔

    (code25 n).b = 1 ∧ (code25 n).q ≠ 0 ∧

    ∀ p ∈ (Finset.range (Nat.sqrt n + 1)).filter Nat.Prime,

      sigma n p ≠ 0 := by

  constructor

  · -- (→) Простое ⇒ код без теней + все σ_p ≠ 0

    intro hp

    have h25 := prime_code_no_shadows n hp (by omega)

    have h_sigma : ∀ p ∈ (Finset.range (Nat.sqrt n + 1)).filter Nat.Prime, sigma n p ≠ 0 := by

      intro p hp_in

      by_contra h0

      have hdvd : p ∣ n := by

        unfold sigma at h0

        split_ifs at h0 <;> simp [Nat.dvd_of_mod_eq_zero] at h0

      have h_le : p ≤ Nat.sqrt n := by

        rw [Finset.mem_filter, Finset.mem_range] at hp_in

        exact hp_in.1

      have h_eq : p = n := by

        exact (Nat.dvd_prime hp).mp hdvd |>.resolve_left (Nat.Prime.one_lt hp_in.2 |>.ne')

      nlinarith [Nat.sqrt_lt_self (by omega : n > 1)]

    exact ⟨h25.1, h25.2, h_sigma⟩

  · -- (←) Код без теней + σ_p ≠ 0 ⇒ простое

    rintro ⟨h_b, h_q, h_sigma⟩

    by_contra hcomp

    rcases Nat.exists_prime_and_dvd (by omega : n ≠ 1) with ⟨p, hp, hdvd⟩

    have hp_le : p ≤ Nat.sqrt n := by

      by_contra h_gt

      have h_pp : p * p > n := by nlinarith

      have h_le : p ≤ n / p := Nat.le_of_dvd (by omega) hdvd

      nlinarith

    have hp_in : p ∈ (Finset.range (Nat.sqrt n + 1)).filter Nat.Prime := by

      simp [Finset.mem_filter, Finset.mem_range, hp, hp_le]

    have h0 : sigma n p = 0 := by

      unfold sigma

      simp [Nat.mod_eq_zero_of_dvd hdvd]

    exact h_sigma p hp_in h0

/-- Фазовая координата σ_p(n) -/

def sigma (n p : ℕ) : ℚ :=

if p = 0 then 0 else (n % p : ℚ) / (p : ℚ)

/-- Глубина пути до √n -/

def pathDepth (n : ℕ) : ℕ :=

((Finset.range (Nat.sqrt n + 1)).filter Nat.Prime).card

/-- Индекс середины пути -/

def midIndex (n : ℕ) : ℕ := pathDepth n / 2

/-- Точка бифуркации: префикс до p_mid + список допустимых продолжений -/

structure BifurcationPoint where

prefix : List (ℕ × ℕ)  -- (p, остаток) до p_mid

branches : List (List (ℕ × ℕ))  -- допустимые хвосты

count_eq : branch_count = branches.length

/-- Фактор ветвления B(n) -/

noncomputable def branchingFactor (n : ℕ) : ℕ :=

let mid := midIndex n

let primes := ((Finset.range (Nat.sqrt n + 1)).filter Nat.Prime).toList

let suffix := primes.drop mid

suffix.foldl (fun acc p => acc * (p - 1)) 1

/-- Узел дерева с явной бифуркацией -/

inductive PhaseTreeNode

| leaf : PhaseTreeNode

| linear (p : ℕ) (res : ℕ) (child : PhaseTreeNode) : PhaseTreeNode

| bifurcate (mid_p : ℕ) (branches : List PhaseTreeNode) : PhaseTreeNode

/-- Подсчёт количества терминальных ветвей в дереве -/

def PhaseTreeNode.branchCount : ℕ

| .leaf => 1

| .linear _ _ child => child.branchCount

| .bifurcate _ branches => branches.foldl (fun acc b => acc + b.branchCount) 0

/-- Вспомогательная функция для построения линейного суффикса -/

| [], _ => PhaseTreeNode.leaf

| p :: ps, n =>


let res := n % p

PhaseTreeNode.linear p res (buildLinearSuffix ps n)
